111 public String getCustomersString() {
112 WebTarget DeustoCarsWebTarget = webTarget.path("server/getcustomers");
113 Invocation.Builder invocationBuilder = DeustoCarsWebTarget.request(MediaType.APPLICATION_JSON);
114 Response response = invocationBuilder.get();
115 if (response.getStatus() != Status.OK.getStatusCode()) {
116 logger.error("Error connecting with the server. Code: {}", response.getStatus());
117 return "Error";
118 } else {
119 String responseMessage = response.readEntity(String.class);
120 logger.info("* Customers coming from the server: '{}'", responseMessage);
121 return responseMessage;
