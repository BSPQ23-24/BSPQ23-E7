157 private void registerVehicle() {
158 // Create the new vehicle
159 VehicleData newVehicle = new VehicleData(numberPlateField.getText(), brandField.getText(), modelField.getText(),
160 readyToBorrowCheckbox.isSelected(), onRepairCheckbox.isSelected());
161
162 WebTarget DeustoCarsWebTarget = ServiceLocator.getInstance().getWebTarget().path("server/addvehicle");
163 Invocation.Builder invocationBuilder = DeustoCarsWebTarget.request(MediaType.APPLICATION_JSON);
164 Response response = invocationBuilder.post(Entity.entity(newVehicle, MediaType.APPLICATION_JSON));
165 if (response.getStatus() != Status.OK.getStatusCode()) {
166 logger.error("Error connecting with the server. Code: {}", response.getStatus());
167 } else {
168 logger.info("Vehicle Correctly Registered :)");

171 }
