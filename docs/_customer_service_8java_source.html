48 public Response addCustomer(CustomerData customerData) {
49 pm = pmf.getPersistenceManager();
50 tx = pm.currentTransaction();
51 try {
52 logger.info("Adding customer...");
53 tx.begin(); // Begin transaction
54 CustomerJDO customer = null;
55 try {
56 customer = pm.getObjectById(CustomerJDO.class, customerData.geteMail());
57 } catch (Exception e) {
58 logger.info("New customer will be added to database");
59 }
60 logger.info("Customer: {}", customer);
61 if (customer != null) {
62 logger.info("Customer with that email already exists, editing entry...");
63 customer.setName(customerData.getName());
64 customer.setSurname(customerData.getSurname());
65 customer.setDateOfBirth(customerData.getDateOfBirth());
66 pm.makePersistent(customer); // Persist the customer object
67 tx.commit(); // Commit the transaction
68 return Response.ok().entity("Customer with that email already exists, entry edited.").build();
69 } else {
70 logger.info("Creating customer: {}", customer);
71 customer = CustomerAssembler.getInstance().CustomerDataToJDO(customerData);
72 pm.makePersistent(customer); // Persist the customer object
73 logger.info("Customer added: {}", customer);
74 }
75 tx.commit(); // Commit the transaction
76 return Response.ok().build();
77 } catch (Exception e) {
78 if (tx.isActive()) {
79 tx.rollback(); // Rollback the transaction in case of an exception
80 }
81 e.printStackTrace();
82 return Response.status(Response.Status.INTERNAL_SERVER_ERROR).entity("Error occurred while adding customer.").build();

84 logger.info("Closing");

148 CustomerJDO customer = pm.getObjectById(CustomerJDO.class, email);

152 for(RentingJDO rent : (List<RentingJDO>) RentingAssembler.getInstance().RentingListToJDO((List<Renting>) RentingService.getInstance().getCustomerRents(customer.geteMail()).getEntity())){
153 RentingService.getInstance().deleteRenting(rent.getVehicle().getNumberPlate(), rent.getCustomer().geteMail());
154 }
155 } catch (Exception e) {
156 logger.info("Customer has no rents asociated");

158 pm.deletePersistent(customer);

162 }
163 } catch (JDOObjectNotFoundException e) {
