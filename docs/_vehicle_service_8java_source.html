48 public Response addVehicle(VehicleData vehicleData) {
49 pm = pmf.getPersistenceManager();
50 tx = pm.currentTransaction();
51 try {
52 logger.info("Adding vehicle...");
53 tx.begin(); // Begin transaction
54 VehicleJDO vehicle = null;
55 try {
56 vehicle = pm.getObjectById(VehicleJDO.class, vehicleData.getNumberPlate());
57 } catch (Exception e) {
58 logger.info("New vehicle will be added to database");
59 }
60 logger.info("Vehicle: {}", vehicle);
61 if (vehicle != null) {
62 logger.info("Vehicle with that number plate already exists, editing entry...");
63 vehicle.setBrand(vehicleData.getBrand());
64 vehicle.setModel(vehicleData.getModel());
65 vehicle.setOnRepair(vehicleData.isOnRepair());
66 vehicle.setReadyToBorrow(vehicleData.isReadyToBorrow());
67 pm.makePersistent(vehicle); // Persist the vehicle object
68 tx.commit(); // Commit the transaction
69 return Response.ok().entity("Vehicle with that number plate already exists, entry edited.").build();
70 } else {
71 logger.info("Creating vehicle: {}", vehicle);
72 vehicle = VehicleAssembler.getInstance().VehicleDataToJDO(vehicleData);
73
74 pm.makePersistent(vehicle); // Persist the vehicle object
75 logger.info("Vehicle added: {}", vehicle);
76 }
77 tx.commit(); // Commit the transaction
78 return Response.ok().build();
79 } catch (Exception e) {
80 if (tx.isActive()) {
81 tx.rollback(); // Rollback the transaction in case of an exception
82 }
83 e.printStackTrace();
84 return Response.status(Response.Status.INTERNAL_SERVER_ERROR).entity("Error occurred while adding vehicle.").build();
85 } finally {
86 logger.info("Closing");
87 pm.close(); // Close the PersistenceManager
88 }

148 tx = pm.currentTransaction();
149 try {
150 VehicleJDO vehicle = pm.getObjectById(VehicleJDO.class, numberPlate);
151 if (vehicle != null) {
152 try {
153 for(RentingJDO rent : (List<RentingJDO>) RentingAssembler.getInstance().RentingListToJDO((List<Renting>) RentingService.getInstance().getVehicleRents(vehicle.getNumberPlate()).getEntity())){
154 RentingService.getInstance().deleteRenting(rent.getVehicle().getNumberPlate(), rent.getCustomer().geteMail());
155 }
156 } catch (Exception e) {
157 logger.info("Vehicle has no rents asociated");
158 }
159 pm.deletePersistent(vehicle);
160 return Response.ok("Vehicle deleted successfully").build();
161 } else {
162 return Response.status(Response.Status.NOT_FOUND).entity("Vehicle not found").build();
163 }
164 } catch (JDOObjectNotFoundException e) {
165 logger.info("Vehicle not found: {}", numberPlate);
