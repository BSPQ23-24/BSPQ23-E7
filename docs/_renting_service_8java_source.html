46 public Response addRenting(Renting renting) {
47 pm = pmf.getPersistenceManager();
48 tx = pm.currentTransaction();
49 try {
50 logger.info("Searching for vehicle to rent...");
51 tx.begin(); // Begin transaction
52 VehicleJDO vehicle = null;
53 try {
54 vehicle = pm.getObjectById(VehicleJDO.class, renting.getVehicle().getNumberPlate());
55 } catch (Exception e) {
56 logger.info("Vehicle not available to rent");
57 return Response.status(Response.Status.INTERNAL_SERVER_ERROR).entity("Vehicle with that number plate does not exists.").build();
58 }
59 logger.info("Vehicle: {}", vehicle);
60 if (vehicle == null) {
61 logger.info("Vehicle not available to rent");
62 return Response.status(Response.Status.INTERNAL_SERVER_ERROR).entity("Vehicle with that number plate does not exists.").build();
63 } else {
64 logger.info("Retrieved vehicle for Renting: {}", vehicle);
65 if (!vehicle.isReadyToBorrow()) {
66 logger.info("Retrieved vehicle is not Ready to borrow: {}", vehicle);
67 return Response.status(Response.Status.INTERNAL_SERVER_ERROR).entity("Vehicle with that number plate is not ready to borrow.").build();
68 }
69 }
70 CustomerJDO customer = null;
71 try {
72 customer = pm.getObjectById(CustomerJDO.class, renting.getCustomer().geteMail());
73 } catch (Exception e) {
74 logger.info("Customer not available to rent");
75 return Response.status(Response.Status.INTERNAL_SERVER_ERROR).entity("Customer with that eMail does not exist.").build();
76 }
77 logger.info("Customer: {}", customer);
78 if (customer == null) {
79 logger.info("Customer not available to rent");
80 return Response.status(Response.Status.INTERNAL_SERVER_ERROR).entity("Customer with that eMail does not exist.").build();
81 } else {
82 logger.info("Retrieved customer for Renting: {}", customer);
83 }
84
85 vehicle.setReadyToBorrow(false);
86 pm.makePersistent(vehicle);
87 RentingJDO rentToSave = new RentingJDO(customer, vehicle, renting.getStartDate(), renting.getEndDate());
88 pm.makePersistent(rentToSave);
89
90 tx.commit(); // Commit the transaction
91 return Response.ok().build();
92 } catch (Exception e) {
93 if (tx.isActive()) {
94 tx.rollback(); // Rollback the transaction in case of an exception
95 }
96 e.printStackTrace();
97 return Response.status(Response.Status.BAD_REQUEST).entity("Error occurred while adding a renting.").build();
98 } finally {
99 logger.info("Closing");
100 pm.close(); // Close the PersistenceManager
101 }
102 }
