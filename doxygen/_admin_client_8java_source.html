175  private void updateDatasetFromServer(XYSeriesCollection dataset) {
176  dataset.removeAllSeries();
177  List<VehicleData> vehicles = VehicleController.getVehicles();
178  List<Renting> rentings = new ArrayList<>();
179  for (VehicleData vehicle : vehicles) {
180  rentings.addAll(RentingController.getVehicleRents(vehicle.getNumberPlate()));
181  }
182  Map<Integer, Integer> rentsPerYear = new TreeMap<>();
183  SimpleDateFormat yearFormat = new SimpleDateFormat("yyyy");
184  for (Renting rent : rentings) {
185  int year = Integer.parseInt(yearFormat.format(rent.getStartDate()));
186  rentsPerYear.merge(year, 1, Integer::sum);
187  }
188 
189  XYSeries series = new XYSeries("Vehicles rented");
190  rentsPerYear.forEach(series::add);
191  dataset.addSeries(series);
192  }
